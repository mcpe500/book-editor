# SPEC: Agentic Book Editor - Multi-Provider AI + MCP Architecture

**Version:** 3.0.0
**Created:** 2026-04-16
**Project:** Book Editor untuk Academic Thesis Writing
**Architecture:** React + Vite (Frontend) + Express.js (Backend) + JSON Files (Database) + Multi-Provider AI + MCP

---

## 1. Executive Summary

### 1.1 Background

Evolution dari arsitektur React + Express menjadi **Agentic Book Editor** dengan dukungan multi-provider AI dan MCP (Model Context Protocol). User request: AI tidak hanya untuk chat, tapi bisa execute actions via tools.

### 1.2 Key Capabilities

| Capability | Description |
|------------|-------------|
| **Multi-Provider AI** | NVIDIA, Gemini, OpenAI, Anthropic, Ollama |
| **MCP Integration** | Built-in + external MCP servers |
| **Agentic AI** | AI can take actions, not just chat |
| **Tool Calling** | Standardized function calling via MCP |
| **Approval Workflow** | User controls AI edit actions |
| **Research** | AI can search web, find references |
| **Writing** | AI can write/edit book content |
| **Review** | AI can review grammar, consistency |

### 1.3 Architecture Decisions

| Component | Decision | Rationale |
|-----------|----------|-----------|
| AI Providers | NVIDIA, Gemini, OpenAI, Anthropic, Ollama | Flexibility, different capabilities |
| MCP Protocol | Standard MCP 2024-11-05 | Industry standard, tool interoperability |
| Built-in MCPs | book-tools, web-tools, file-tools | Core functionality |
| External MCP | Supported (stdio transport) | Extensibility |
| Agent Autonomy | Semi-autonomous | Balance capability + control |
| API Key Storage | Encrypted in backend | Security |

```typescript
export interface Message {
  role: 'system' | 'user' | 'assistant';
  content: string;
}

export interface ChatOptions {
  temperature?: number;
  maxTokens?: number;
  topP?: number;
  tools?: Tool[];

  name: string;
  description: string;
  parameters: Record<string, unknown>;
}

export interface ChatResponse {
  content: string;
  toolCalls?: ToolCall[];
  finishReason: 'stop' | 'length' | 'tool_calls' | 'error';

export interface ToolCall {
  id: string;
  name: string;
  arguments: Record<string, unknown>;
}

export interface AIProvider {
  readonly id: string;
  readonly name: string;
  readonly displayName: string;
  readonly supportsTools: boolean;
  readonly supportsVision: boolean;
  readonly maxContextWindow: number;

  chat(messages: Message[], options?: ChatOptions): Promise<ChatResponse>;
  streamChat(
    messages: Message[],
    options: ChatOptions,

  ): Promise<void>;
}
```

```typescript
// src/services/ai/providers/nvidia.ts

export class NvidiaProvider implements AIProvider {
  readonly id = 'nvidia';
  readonly name = 'nvidia';
  readonly displayName = 'NVIDIA';
  readonly supportsTools = false; // minimax-m2.7 limited tool support
  readonly supportsVision = false;
  readonly maxContextWindow = 8192;

  async chat(messages: Message[], options?: ChatOptions): Promise<ChatResponse> {
    const response = await fetch('https://integrate.api.nvidia.com/v1/chat/completions', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: 'minimaxai/minimax-m2.7',
        messages,
        temperature: options?.temperature ?? 1,
        max_tokens: options?.maxTokens ?? 8192,
        top_p: options?.topP ?? 0.95,
      }),
    });

    const data = await response.json();
    return {
      content: data.choices[0].message.content,
      finishReason: 'stop',
    };
  }

  async streamChat(
    messages: Message[],
    options: ChatOptions,
    onChunk: (chunk: string) => void
  ): Promise<void> {
    // SSE streaming implementation
    const response = await fetch('https://integrate.api.nvidia.com/v1/chat/completions', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: 'minimaxai/minimax-m2.7',
        messages,
        stream: true,
        temperature: options?.temperature ?? 1,
        max_tokens: options?.maxTokens ?? 8192,
      }),
    });

    const reader = response.body?.getReader();
    const decoder = new TextDecoder();

    while (true) {
      const { done, value } = await reader!.read();
      if (done) break;

      const chunk = decoder.decode(value);
      const lines = chunk.split('\n');
      for (const line of lines) {
        if (line.startsWith('data: ')) {
          const data = JSON.parse(line.slice(6));
          if (data.choices?.[0]?.delta?.content) {
            onChunk(data.choices[0].delta.content);
          }
        }
      }
    }
  }
}
```

#### Gemini Provider

```typescript
// src/services/ai/providers/gemini.ts

export class GeminiProvider implements AIProvider {
  readonly id = 'gemini';
  readonly name = 'gemini';
  readonly displayName = 'Google Gemini';
  readonly supportsTools = true;
  readonly supportsVision = true;
  readonly maxContextWindow = 1000000; // 1M tokens for Gemini 1.5

  private getModel(): string {
    // User selectable: gemini-1.5-pro, gemini-1.5-flash, gemma-3-27b
    return this.model ?? 'gemini-2.0-flash';
  }

  async chat(messages: Message[], options?: ChatOptions): Promise<ChatResponse> {
    const contents = messages.map(m => ({
      role: m.role === 'assistant' ? 'model' : 'user',
      parts: [{ text: m.content }],
    }));

    const requestBody: any = {
      contents,
      generationConfig: {
        temperature: options?.temperature ?? 0.9,
        maxOutputTokens: options?.maxTokens ?? 8192,
        topP: options?.topP ?? 1,
      },
    };

    if (options?.tools) {
      requestBody.tools = {
        functionDeclarations: options.tools.map(t => ({
          name: t.name,
          description: t.description,
          parameters: {
            type: 'object',
            properties: Object.fromEntries(
              Object.entries(t.parameters).map(([k, v]) => [k, { ...v }])
            ),
          },
        })),
      };
    }

    const response = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/${this.getModel()}:generateContent?key=${this.apiKey}`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(requestBody),
      }
    );

    const data = await response.json();
    const candidate = data.candidates?.[0];
    const content = candidate?.content?.parts?.[0];

    if (content?.functionCall) {
      return {
        content: '',
        toolCalls: [{
          id: crypto.randomUUID(),
          name: content.functionCall.name,
          arguments: content.functionCall.args,
        }],
        finishReason: 'tool_calls',
      };
    }

    return {
      content: content?.text ?? '',
      finishReason: 'stop',
    };
  }

  async streamChat(
    messages: Message[],
    options: ChatOptions,
    onChunk: (chunk: string) => void
  ): Promise<void> {
    // Streaming implementation via SSE
    // Similar pattern to NVIDIA but with Gemini's streaming endpoint
  }
}
```

```typescript
export class OpenAIProvider implements AIProvider {
  readonly id = 'openai';
  readonly name = 'openai';
  readonly displayName = 'OpenAI';
  readonly supportsTools = true;
  readonly supportsVision = true;
  readonly maxContextWindow = 128000; // GPT-4 Turbo

  async chat(messages: Message[], options?: ChatOptions): Promise<ChatResponse> {
    const response = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: this.model ?? 'gpt-4-turbo',
        messages,
        temperature: options?.temperature ?? 0.7,
        max_tokens: options?.maxTokens ?? 4096,
        top_p: options?.topP ?? 1,
        tools: options?.tools?.map(t => ({
          type: 'function',
          function: {
            name: t.name,
            description: t.description,
            parameters: {
              type: 'object',
              properties: Object.fromEntries(
                Object.entries(t.parameters).map(([k, v]) => [k, { ...v }])
              ),
            },
          },
        })),
        tool_choice: options?.toolChoice,
      }),
    });

    const data = await response.json();
    const choice = data.choices[0];

    if (choice.message.tool_calls) {
      return {
        content: choice.message.content ?? '',
        toolCalls: choice.message.tool_calls.map((tc: any) => ({
          id: tc.id,
          name: tc.function.name,
          arguments: JSON.parse(tc.function.arguments),
        })),
        finishReason: 'tool_calls',
      };
    }

    return {
      content: choice.message.content,
      finishReason: 'stop',
    };
  }
}
```

### 3.3 Provider Manager

```typescript
// src/services/ai/providerManager.ts

export class AIProviderManager {
  private providers: Map<string, AIProvider> = new Map();
  private configs: Map<string, AIProviderConfig> = new Map();

  async initialize(): Promise<void> {
    // Load configs from data/ai-providers.json
    const configs = await readJSON<AIProviderConfig[]>('data/ai-providers.json');

    for (const config of configs) {
      if (!config.enabled) continue;

      const provider = this.createProvider(config);
      if (provider) {
        this.providers.set(config.id, provider);
        this.configs.set(config.id, config);
      }
    }
  }

  private createProvider(config: AIProviderConfig): AIProvider | null {
    switch (config.provider) {
      case 'nvidia':
        return new NvidiaProvider(config.apiKey);
      case 'gemini':
        return new GeminiProvider(config.apiKey, config.model);
      case 'openai':
        return new OpenAIProvider(config.apiKey, config.model);
      case 'anthropic':
        return new AnthropicProvider(config.apiKey, config.model);
      case 'ollama':
        return new OllamaProvider(config.baseUrl ?? 'http://localhost:11434', config.model);
      default:
        return null;
    }
  }

  getProvider(id: string): AIProvider | null {
    return this.providers.get(id) ?? null;
  }

  getProviders(): AIProviderConfig[] {
    return Array.from(this.configs.values());
  }

  async saveProvider(config: AIProviderConfig): Promise<void> {
    const configs = await readJSON<AIProviderConfig[]>('data/ai-providers.json');
    const index = configs.findIndex(c => c.id === config.id);

    if (index >= 0) {
      configs[index] = config;
    } else {
      configs.push(config);
    }

    await writeJSON('data/ai-providers.json', configs);
    this.configs.set(config.id, config);

    const provider = this.createProvider(config);
    if (provider) {
      this.providers.set(config.id, provider);
    }
  }
}

export const providerManager = new AIProviderManager();
```

```typescript
  async deleteChapter(bookId: string, chapterId: string) {
    const book = await readJSON<Book>(`data/books/${bookId}.json`);
    const index = book.chapters.findIndex(c => c.id === chapterId);
    if (index === -1) {
      return { content: [{ type: 'text', text: 'Chapter not found' }], isError: true };
    }
    book.chapters.splice(index, 1);
    await writeJSON(`data/books/${bookId}.json`, book);
    return { content: [{ type: 'text', text: 'Chapter deleted' }] };

  async searchBook(bookId: string, query: string) {
    const book = await readJSON<Book>(`data/books/${bookId}.json`);
    const results: Array<{ chapterId: string; chapterTitle: string; matches: string[] }> = [];

    for (const chapter of book.chapters) {
      const matches: string[] = [];
      const lines = chapter.content.split('\n');
      for (const line of lines) {
        if (line.toLowerCase().includes(query.toLowerCase())) {
          matches.push(line.trim().substring(0, 200));
        }
      }
      if (matches.length > 0) {
        results.push({
          chapterId: chapter.id,
          chapterTitle: chapter.title,
          matches,
        });
      }
    }

    return { content: [{ type: 'text', text: JSON.stringify(results, null, 2) }] };
  },

    if (chapterId) {
      const chapter = book.chapters.find(c => c.id === chapterId);
      const count = chapter?.content.split(/\s+/).filter(Boolean).length ?? 0;
      return { content: [{ type: 'text', text: `Word count: ${count}` }] };
    }
    const total = book.chapters.reduce((sum, c) => sum + c.content.split(/\s+/).filter(Boolean).length, 0);
    return { content: [{ type: 'text', text: `Total word count: ${total}` }] };
  },
```

```typescript
export class MCPToolExecutor {
  private builtInHandlers: Record<string, Record<string, Function>> = {
    'book-tools': bookToolsHandlers,
    'web-tools': webToolsHandlers,
  };

  private externalClients: Map<string, MCPClient> = new Map();

  async execute(
    serverName: string,
    toolName: string,
    args: any,
    context: { bookId: string; chapterId?: string }
  ): Promise<MCPToolResult> {
    // Check built-in handlers first
    const builtIn = this.builtInHandlers[serverName];
    if (builtIn && builtIn[toolName]) {
      return await builtIn[toolName](context.bookId, context.chapterId, ...Object.values(args));
    }

    // Check external MCP servers
    const client = this.externalClients.get(serverName);
    if (client) {
      return await client.executeTool(serverName, toolName, args);
    }

    throw new Error(`Tool ${toolName} not found on server ${serverName}`);
  }

  registerExternalServer(serverId: string, client: MCPClient): void {
    this.externalClients.set(serverId, client);
  }
}
```

### 5.2 Agent Engine

```typescript
// src/services/agents/agentEngine.ts

export class AgentEngine {
  private providerManager: AIProviderManager;
  private toolExecutor: MCPToolExecutor;

  async startRun(run: AgentRun, config: AgentConfig): Promise<void> {
    // 1. Initialize run state
    run.state = 'running';
    run.startedAt = new Date().toISOString();
    await this.saveRun(run);

    // 2. Build system prompt with context
    const systemPrompt = this.buildSystemPrompt(config, run.context);
    const messages: Message[] = [
      { role: 'system', content: systemPrompt },
      { role: 'user', content: run.context.currentChapter ?? 'No current chapter' },
    ];

    // 3. Main agent loop
    while (run.state === 'running' && run.tasks.length > 0) {
      const currentTask = run.tasks[0];

      if (currentTask.status === 'pending') {
        currentTask.status = 'in_progress';
        currentTask.startedAt = new Date().toISOString();
        await this.saveRun(run);
      }

      // 4. Send to AI with current context
      const provider = this.providerManager.getProvider(config.providerId);
      if (!provider) {
        run.state = 'failed';
        run.currentTask = currentTask;
        await this.saveRun(run);
        break;
      }

      // 5. Get AI response (may include tool calls)
      const response = await provider.chat(messages, {
        temperature: config.temperature,
        maxTokens: config.maxTokens,
        tools: this.getAvailableTools(config.mcpServerIds),
      });

      // 6. Add response to conversation
      messages.push({ role: 'assistant', content: response.content });

      // 7. Handle tool calls
      if (response.toolCalls) {
        for (const toolCall of response.toolCalls) {
          const execution: ToolCallExecution = {
            id: crypto.randomUUID(),
            toolName: toolCall.name,
            arguments: toolCall.arguments,
            status: 'pending',
          };

          // Check if auto-approve
          if (config.autoApproveTools.includes(toolCall.name)) {
            execution.status = 'approved';
            execution.approvedAt = new Date().toISOString();
          } else {
            execution.status = 'awaiting_approval';
            currentTask.toolCalls.push(execution);
            currentTask.status = 'awaiting_approval';
            await this.saveRun(run);
            // Wait for user approval
            await this.waitForApproval(run.id, execution.id);
          }

          if (execution.status === 'approved') {
            await this.executeToolCall(execution, run);
          }
        }
      }

      // 8. Check if task complete
      if (currentTask.status === 'in_progress' && !response.toolCalls) {
        currentTask.status = 'completed';
        currentTask.completedAt = new Date().toISOString();
        currentTask.result = response.content;
        run.tasks.shift(); // Move to next task
      }

      await this.saveRun(run);
    }

    // 9. Finalize
    if (run.state === 'running') {
      run.state = 'completed';
    }
    run.completedAt = new Date().toISOString();
    await this.saveRun(run);
  }

  private async executeToolCall(execution: ToolCallExecution, run: AgentRun): Promise<void> {
    execution.status = 'executing';

    try {
      const [serverName, toolName] = execution.toolName.split('.');
      const result = await this.toolExecutor.execute(
        serverName,
        toolName,
        execution.arguments,
        { bookId: run.bookId, chapterId: run.chapterId }
      );

      execution.result = result;
      execution.status = 'completed';
    } catch (error) {
      execution.error = error.message;
      execution.status = 'failed';
    }
  }

  private async waitForApproval(runId: string, executionId: string): Promise<void> {
    // Poll for approval status change
    // This would be implemented with events or polling
    while (true) {
      const run = await this.loadRun(runId);
      const execution = run.tasks[0]?.toolCalls.find(tc => tc.id === executionId);

      if (execution?.status === 'approved' || execution?.status === 'rejected') {
        return;
      }

      await new Promise(resolve => setTimeout(resolve, 1000));
    }
  }

  private buildSystemPrompt(config: AgentConfig, context: any): string {
    let prompt = config.systemPrompt;

    prompt += `\n\nAvailable tools: ${config.mcpServerIds.join(', ')}`;
    prompt += `\nCurrent book ID: ${context.bookId}`;

    if (context.bookOutline) {
      prompt += `\n\nBook outline:\n${context.bookOutline}`;
    }

    if (context.selectedText) {
      prompt += `\n\nSelected text:\n${context.selectedText}`;
    }

    return prompt;
  }

  private getAvailableTools(mcpServerIds: string[]): Tool[] {
    const tools: Tool[] = [];

    for (const serverId of mcpServerIds) {
      if (serverId === 'book-tools') {
        for (const tool of bookToolsMCPServer.tools) {
          tools.push({
            name: `book-tools.${tool.name}`,
            description: tool.description,
            parameters: tool.inputSchema as any,
          });
        }
      } else if (serverId === 'web-tools') {
        for (const tool of webToolsMCPServer.tools) {
          tools.push({
            name: `web-tools.${tool.name}`,
            description: tool.description,
            parameters: tool.inputSchema as any,
          });
        }
      }
    }

    return tools;
  }
}
```

## 9. Security

### 9.1 API Key Encryption

```typescript
// src/utils/crypto.ts
import { createCipheriv, createDecipheriv, randomBytes } from 'crypto';

const ENCRYPTION_KEY = process.env.ENCRYPTION_KEY; // 32 bytes
const ALGORITHM = 'aes-256-gcm';

export function encryptApiKey(apiKey: string): string {
  const iv = randomBytes(16);
  const cipher = createCipheriv(ALGORITHM, Buffer.from(ENCRYPTION_KEY!, 'hex'), iv);

  let encrypted = cipher.update(apiKey, 'utf8', 'hex');
  encrypted += cipher.final('hex');

  const authTag = cipher.getAuthTag();

  return `${iv.toString('hex')}:${authTag.toString('hex')}:${encrypted}`;
}

export function decryptApiKey(encrypted: string): string {
  const [ivHex, authTagHex, encryptedData] = encrypted.split(':');

  const iv = Buffer.from(ivHex, 'hex');
  const authTag = Buffer.from(authTagHex, 'hex');
  const decipher = createDecipheriv(ALGORITHM, Buffer.from(ENCRYPTION_KEY!, 'hex'), iv);
  decipher.setAuthTag(authTag);

  let decrypted = decipher.update(encryptedData, 'hex', 'utf8');
  decrypted += decipher.final('utf8');

  return decrypted;
}
```
