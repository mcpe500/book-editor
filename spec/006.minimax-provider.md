## 2. MiniMax API Details

### 2.1 API Configuration

| Property | Value |
|----------|-------|
| **Endpoint** | `https://api.minimax.chat/v1/text/chatcompletion_v2` |
| **Auth** | `Authorization: Bearer {api_key}` |
| **Default Model** | `MiniMax-M2.7` |
| **Format** | OpenAI-compatible (messages array, SSE streaming) |
| **Supports Streaming** | Yes (SSE) |
| **Supports Tools** | Yes (function calling) |

### 2.2 Request Format

```json
POST /v1/text/chatcompletion_v2
Authorization: Bearer {MINIMAX_API_KEY}
Content-Type: application/json

{
  "model": "MiniMax-M2.7",
  "messages": [
    {"role": "system", "content": "You are helpful."},
    {"role": "user", "content": "Hello"}
  ],
  "temperature": 0.7,
  "max_tokens": 4096,
  "stream": true
}
```

### 2.3 Streaming Response Format

```
data: {"id":"...","choices":[{"delta":{"content":"Hello"}}]}
data: {"id":"...","choices":[{"delta":{"content":" world"}}]}
data: [DONE]
```

#### `backend/src/services/ai/providers/minimax.ts` (NEW FILE)

```typescript
import { Message, ChatOptions, ChatResponse, Tool } from '../../../types/index.js';

export class MiniMaxProvider {
  readonly id = 'minimax';
  readonly name = 'minimax';
  readonly displayName = 'MiniMax AI';
  readonly supportsTools = true;
  readonly supportsVision = false;
  readonly maxContextWindow = 1000000; // 1M tokens

  private apiKey: string;
  private model: string;

  constructor(apiKey: string, model?: string) {
    this.apiKey = apiKey;
    this.model = model ?? 'MiniMax-M2.7';
  }

  private getApiUrl(): string {
    return 'https://api.minimax.chat/v1/text/chatcompletion_v2';
  }

  async chat(messages: Message[], options?: ChatOptions): Promise<ChatResponse> {
    const body: Record<string, unknown> = {
      model: this.model,
      messages,
      temperature: options?.temperature ?? 0.7,
      max_tokens: options?.maxTokens ?? 4096,
    };

    if (options?.tools) {
      body.tools = options.tools.map((t: Tool) => ({
        type: 'function',
        function: {
          name: t.name,
          description: t.description,
          parameters: {
            type: 'object',
            properties: Object.fromEntries(
              Object.entries(t.parameters).map(([k, v]) => [k, { ...v as object }])
            ),
          },
        },
      }));
      body.tool_choice = options.toolChoice ?? 'auto';
    }

    const response = await fetch(this.getApiUrl(), {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(body),
    });

    if (!response.ok) {
      throw new Error(`MiniMax API error: ${response.status}`);
    }

    const data = await response.json() as {
      choices: Array<{
        message?: { content?: string; function_call?: { name: string; arguments: string } };
      }>;
    };
    const choice = data.choices?.[0];

    if (choice?.message?.function_call) {
      return {
        content: '',
        toolCalls: [{
          id: crypto.randomUUID(),
          name: choice.message.function_call.name,
          arguments: JSON.parse(choice.message.function_call.arguments),
        }],
        finishReason: 'tool_calls',
      };
    }

    return {
      content: choice?.message?.content ?? '',
      finishReason: 'stop',
    };
  }

  async streamChat(
    messages: Message[],
    options: ChatOptions,
    onChunk: (chunk: string) => void,
    onComplete: () => void,
    onError: (error: Error) => void
  ): Promise<void> {
    const body: Record<string, unknown> = {
      model: this.model,
      messages,
      stream: true,
      temperature: options?.temperature ?? 0.7,
      max_tokens: options?.maxTokens ?? 4096,
    };

    if (options?.tools) {
      body.tools = options.tools.map((t: Tool) => ({
        type: 'function',
        function: {
          name: t.name,
          description: t.description,
          parameters: {
            type: 'object',
            properties: Object.fromEntries(
              Object.entries(t.parameters).map(([k, v]) => [k, { ...v as object }])
            ),
          },
        },
      }));
      body.tool_choice = options.toolChoice ?? 'auto';
    }

    try {
      const response = await fetch(this.getApiUrl(), {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify(body),
      });

      if (!response.ok) {
        throw new Error(`MiniMax API error: ${response.status}`);
      }

      const reader = response.body?.getReader();
      if (!reader) {
        throw new Error('No response body');
      }

      const decoder = new TextDecoder();

      const processStream = async () => {
        while (true) {
          const { done, value } = await reader.read();

          if (done) {
            onComplete();
            break;
          }

          const chunk = decoder.decode(value, { stream: true });
          const lines = chunk.split('\n');

          for (const line of lines) {
            if (line.startsWith('data: ')) {
              const dataStr = line.slice(6);
              if (dataStr === '[DONE]') {
                onComplete();
                return;
              }
              try {
                const data = JSON.parse(dataStr);
                if (data.choices?.[0]?.delta?.content) {
                  onChunk(data.choices[0].delta.content);
                }
              } catch {
                // Skip invalid JSON
              }
            }
          }
        }
      };

      await processStream();
    } catch (error) {
      onError(error as Error);
    }
  }
}
```

export function useProviders() {

  const testNVIDIAConnection = useMutation({

    },
  })

  const testMiniMaxConnection = useMutation({

    },
  })

    currentProvider: settings.aiProvider,
    nvidiaApiKey: settings.nvidiaApiKey,
    minimaxApiKey: settings.minimaxApiKey,
    updateSettings,
    testNVIDIAConnection,
    testMiniMaxConnection,
